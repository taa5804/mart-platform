// ========================================== // 0. URL 파라미터에서 QR 코드 추출 // ========================================== const urlParams = new URLSearchParams(window.location.search); const qrCode = urlParams.get("qr") || ""; // ========================================== // 1. 고유 세션 ID 관리 (LocalStorage) // ========================================== function getOrCreateSessionId() { let sessionId = localStorage.getItem("caller_session_id"); if (!sessionId || sessionId.length < 12) { sessionId = "sess_" + Math.random().toString(36).substring(2, 14); localStorage.setItem("caller_session_id", sessionId); } return sessionId; } const callerSessionId = getOrCreateSessionId(); // ========================================== // 2. 실시간 호출 상태 조회 및 UI 제어 // ========================================== async function checkCallStatus() { const pushButton = document.getElementById("pushButton"); const emergencyButton = document.getElementById("emergencyButton"); const resultMessage = document.getElementById("pushResult"); if (!qrCode) { if (resultMessage) resultMessage.textContent = "오류: 올바르지 않은 QR 접근입니다."; return; } try { const { data, error } = await db.functions.invoke("send-vehicle-push", { body: { action: "call-status", qr_code: qrCode, caller_session_id: callerSessionId } }); if (error || !data || data.ok === false) { console.error("호출 상태 조회 실패:", error || data?.message); return; } if (data.next_call_available) { pushButton.disabled = false; pushButton.textContent = data.call_count === 0 ? "푸시알림 보내기" : `${data.call_count}차 요청 완료 (추가 가능)`; } else { pushButton.disabled = true; if (data.call_count >= 3) { pushButton.textContent = "3차 호출 완료 (호출 마감)"; } else { pushButton.textContent = "대기 시간 제한 (잠시 후 가능)"; } } if (emergencyButton) { if (data.emergency_available) { emergencyButton.style.display = "block"; emergencyButton.disabled = false; emergencyButton.textContent = "🚨 차주와 긴급 통화하기"; } else { if (data.emergency_call_used) { emergencyButton.style.display = "block"; emergencyButton.disabled = true; emergencyButton.textContent = "❌ 긴급통화 사용 완료"; } else if (data.call_count >= 3) { emergencyButton.style.display = "block"; emergencyButton.disabled = true; emergencyButton.textContent = "⏳ 잠시 후 긴급통화 가능"; } else { emergencyButton.style.display = "none"; } } } } catch (err) { console.error("상태 로드 중 네트워크 오류:", err); } } // ========================================== // 3. 차량 이동 알림 전송 // ========================================== async function sendVehicleCall() { const pushButton = document.getElementById("pushButton"); const resultMessage = document.getElementById("pushResult"); if (!qrCode) return alert("QR 코드 정보가 부족합니다."); pushButton.disabled = true; pushButton.textContent = "전송 중..."; resultMessage.textContent = ""; try { const { data, error } = await db.functions.invoke("send-vehicle-push", { body: { action: "call", qr_code: qrCode, caller_session_id: callerSessionId } }); if (error) throw new Error(error.message); if (data && data.ok === false) throw new Error(data.message || "푸시알림 전송 실패"); resultMessage.style.color = "green"; resultMessage.textContent = `🎉 ${data.call_count}차 차량이동 요청이 차주에게 전송되었습니다.`; await checkCallStatus(); } catch (error) { console.error("푸시 전송 실패:", error); resultMessage.style.color = "red"; resultMessage.textContent = "전송 실패: " + (error.message || "알 수 없는 오류"); pushButton.disabled = false; pushButton.textContent = "푸시알림 보내기"; } } // ========================================== // 4. 긴급 통화 다이얼러 연결 // ========================================== async function startEmergencyCall() { const resultMessage = document.getElementById("pushResult"); const callerPhone = prompt("차주와 전화를 연결하기 위해 본인의 휴대전화 번호를 입력해 주세요.\n(숫자만 입력)"); if (!callerPhone) return; if (!qrCode) return alert("QR 코드 정보가 누락되었습니다."); try { const { data, error } = await db.functions.invoke("send-vehicle-push", { body: { action: "emergency-call", qr_code: qrCode, caller_session_id: callerSessionId, caller_phone: callerPhone } }); if (error) throw new Error(error.message); if (data && data.ok === false) throw new Error(data.message || "긴급통화 요청 실패"); resultMessage.style.color = "blue"; resultMessage.textContent = "차주 연락처로 전화를 연결합니다."; if (data.tel_url) window.location.href = data.tel_url; await checkCallStatus(); } catch (error) { alert("오류: " + error.message); } } // ========================================== // 5. 초기 이벤트 바인딩 // ========================================== document.addEventListener("DOMContentLoaded", () => { checkCallStatus(); document.getElementById("pushButton")?.addEventListener("click", sendVehicleCall); document.getElementById("emergencyButton")?.addEventListener("click", startEmergencyCall); });
